    pool_pre_ping=True,
    pool_timeout=10,
    connect_args={

        "command_timeout": 10,
        "server_settings": {
            "statement_timeout": "10000",
            "idle_in_transaction_session_timeout": "30000",
            "application_name": "frgcrm-api-async",
        },
    },
)

# Connection pool rationale:
# - pool_size=10: Enough for typical concurrency. Most API handlers are fast (<100ms).
# - max_overflow=20: Allows temporary bursts without permanent overhead.
# - pool_recycle=3600: Prevents memory bloat; hourly is sufficient.
# - pool_timeout=10: Fail fast — don't make users wait for a connection.
# - connect_timeout=5: If DB is unreachable, fail in 5s instead of OS TCP timeout (30s+).
#
# With 6 uvicorn workers:
#   Regular pool:  6 * 10 = 60 connections
#   Burst pool:    6 * (10+20) = 180 connections max
#   PostgreSQL max: 200 connections
#   Safe margin: 20 connections for maintenance / other services

POOL_MONITORING_SQL = """
-- 1. Current connection count by application
SELECT
    application_name,
    state,
    COUNT(*) AS connections
FROM pg_stat_activity
WHERE application_name != ''
GROUP BY application_name, state
ORDER BY connections DESC;

-- 2. Idle-in-transaction connections (LEAKS)
SELECT
    pid,
    application_name,
    state,
    age(now(), state_change) AS state_duration,
    query
FROM pg_stat_activity
WHERE state = 'idle in transaction'
  AND age(now(), state_change) > interval '30 seconds'
ORDER BY state_change;

-- 3. Long-running queries
SELECT
    pid,
    application_name,
    age(now(), query_start) AS query_duration,
    state,
    left(query, 200) AS query_preview
FROM pg_stat_activity
WHERE state = 'active'
  AND age(now(), query_start) > interval '5 seconds'
ORDER BY query_start;

-- 4. Connection utilization (requires pg_stat_statements extension)
SELECT
    count(*) AS total_connections,
    (SELECT setting::int FROM pg_settings WHERE name = 'max_connections') AS max_connections,
    round(100.0 * count(*) / (SELECT setting::int FROM pg_settings WHERE name = 'max_connections'), 1) AS utilization_pct
FROM pg_stat_activity;

-- 5. Waiting queries (locks)
SELECT
    blocked.pid AS blocked_pid,
    blocked.query AS blocked_query,
    blocking.pid AS blocking_pid,
    blocking.query AS blocking_query,
    age(now(), blocked.query_start) AS wait_duration
FROM pg_stat_activity blocked
JOIN pg_locks blocked_locks ON blocked.pid = blocked_locks.pid
JOIN pg_locks blocking_locks ON blocked_locks.lock_type = blocking_locks.lock_type
    AND blocked_locks.relation = blocking_locks.relation
    AND blocked_locks.pid != blocking_locks.pid
JOIN pg_stat_activity blocking ON blocking_locks.pid = blocking.pid
WHERE NOT blocked_locks.granted;
"""

POOL_HEALTH_CHECKS = """
// Add to ecosystem-guardian/src/validators/postgres.ts or equivalent:

interface PoolHealth {
  service: string;
  poolSize: number;
  activeConnections: number;
  idleConnections: number;
  waitingClients: number;
  utilizationPct: number;
  status: 'healthy' | 'warning' | 'critical';
}

async function checkPoolHealth(pool: Pool, serviceName: string): Promise<PoolHealth> {
  const stats = {
    total: pool.totalCount,
    idle: pool.idleCount,
    waiting: pool.waitingCount,
  };

  const utilizationPct = stats.total > 0
    ? (stats.total - stats.idle) / stats.total * 100
    : 0;

  let status: 'healthy' | 'warning' | 'critical' = 'healthy';
  if (utilizationPct > 90 || stats.waiting > 5) status = 'critical';
  else if (utilizationPct > 70 || stats.waiting > 0) status = 'warning';

  return {
    service: serviceName,
    poolSize: pool.options.max || 10,
    activeConnections: stats.total - stats.idle,
    idleConnections: stats.idle,
    waitingClients: stats.waiting,
    utilizationPct: Math.round(utilizationPct * 10) / 10,
    status,
  };
}
