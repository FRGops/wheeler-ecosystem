## R2: ADD EXPRESS TIMEOUT MIDDLEWARE
# All 9 agent-svc processes lack HTTP timeout configuration.
# Node.js default server.timeout = 0 means connections never timeout.

RECOMMENDED_EXPRESS_TIMEOUT_CONFIG = """
// Add to each agent-svc src/server.ts createServer() function:
export function createServer(svcName, agentRunners) {
  const app = express();
  app.use(express.json());

  // ── REQUEST TIMEOUT MIDDLEWARE ──
  const REQUEST_TIMEOUT_MS = parseInt(process.env.REQUEST_TIMEOUT_MS || '60000', 10);
  app.use((req, res, next) => {
    const timer = setTimeout(() => {
      if (!res.headersSent) {
        res.status(504).json({ error: 'Request timeout', timeout_ms: REQUEST_TIMEOUT_MS });
      }
    }, REQUEST_TIMEOUT_MS);
    res.on('finish', () => clearTimeout(timer));
    res.on('close', () => clearTimeout(timer));
    next();
  });

  // ── BODY SIZE LIMIT ──
  app.use(express.json({ limit: '10mb' }));

  // ... rest of routes ...

  return app;
}

// And in index.ts listen():
const server = app.listen(config.port, () => {
  console.log(`[${svcName}] on :${config.port}`);
});
server.timeout = 30_000;           // 30s idle socket timeout
server.keepAliveTimeout = 65_000;  // Slightly above common LB timeout
server.headersTimeout = 66_000;    // Must be > keepAliveTimeout
server.maxConnections = 200;       // Prevent connection flood
server.requestTimeout = 60_000;    // Node 18+; max request duration
"""

CACHE_IMPLEMENTATION = """
# Python FastAPI example (add to frgcrm-api):
import redis.asyncio as aioredis
import hashlib, json
from functools import wraps

REDIS_URL = os.getenv("REDIS_URL", "redis://localhost:6379")

def redis_cache(ttl_seconds: int = 60):
    def decorator(func):
        @wraps(func)
        async def wrapper(*args, **kwargs):
            try:
                r = aioredis.from_url(REDIS_URL)
                key = f"cache:{func.__module__}.{func.__name__}:{hashlib.sha256(json.dumps(kwargs,sort_keys=True).encode()).hexdigest()[:16]}"
                cached = await r.get(key)
                if cached:
                    return json.loads(cached)
                result = await func(*args, **kwargs)
                await r.setex(key, ttl_seconds, json.dumps(result, default=str))
                return result
            except Exception:
                return await func(*args, **kwargs)  # Fail open
        return wrapper
    return decorator
"""

## R10: ASYNC DATABASE MIGRATION
# Convert frgcrm-api from synchronous SQLAlchemy to async (asyncpg + SQLAlchemy 2.0 async)

ASYNC_DB_EXAMPLE = """
# database_async.py
from sqlalchemy.ext.asyncio import create_async_engine, AsyncSession, async_sessionmaker

DATABASE_URL_ASYNC = os.getenv(
    "DATABASE_URL_ASYNC",
    "postgresql+asyncpg://frgops@127.0.0.1:5433/frgcrm"
)

async_engine = create_async_engine(
    DATABASE_URL_ASYNC,
    pool_size=10,
    max_overflow=20,
    pool_recycle=3600,
    pool_pre_ping=True,
    pool_timeout=10,
)

AsyncSessionLocal = async_sessionmaker(
    async_engine, class_=AsyncSession, expire_on_commit=False
)

async def get_async_db():
    async with AsyncSessionLocal() as session:
        try:
            yield session
            await session.commit()
        except Exception:
            await session.rollback()
            raise
"""
